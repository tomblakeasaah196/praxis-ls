    function renderPaging(total) {

      const label = document.getElementById('paging-label');

      state.page = Math.min(state.page, pages);

      const from = total === 0 ? 0 : (state.page - 1) * state.pageSize + 1;
      const to = Math.min(total, state.page * state.pageSize);
      label.textContent = `Showing ${from}-${to} of ${total}`;

      const mk = (p, text, disabled=false, active=false) => `
        <li class="page-item ${disabled ? 'disabled' : ''} ${active ? 'active' : ''}">
          <a class="page-link ${active ? 'bg-dark border-dark' : 'text-dark'}" href="#" onclick="event.preventDefault(); gotoPage(${p});">${text}</a>
        </li>
      `;

      let html = '';
      html += mk(state.page - 1, 'Prev', state.page <= 1, false);
      for (let p = 1; p <= pages; p++) {
        if (p === 1 || p === pages || Math.abs(p - state.page) <= 1) {
          html += mk(p, String(p), false, p === state.page);
        } else if (Math.abs(p - state.page) === 2) {
          html += `<li class="page-item disabled"><span class="page-link">…</span></li>`;
        }
      }
      html += mk(state.page + 1, 'Next', state.page >= pages, false);
      paging.innerHTML = html;
    }

    // ---- Lines rendering + risk detection
    function isLineRisk(l) {
      const cost = (l.cost_amount_ttc !== null && l.cost_amount_ttc !== undefined) ? Number(l.cost_amount_ttc) : null;
      const sell = (l.sell_amount_ttc !== null && l.sell_amount_ttc !== undefined) ? Number(l.sell_amount_ttc) : null;

      // risk: cost present but sell missing OR sell < cost (when both exist)
      if (cost !== null && cost > 0 && (sell === null || sell === 0)) return true;
      if (cost !== null && sell !== null && sell < cost) return true;
      return false;
    }

    function calcLineMargin(l) {
      const cost = (l.cost_amount_ttc !== null && l.cost_amount_ttc !== undefined) ? Number(l.cost_amount_ttc) : 0;
      const sell = (l.sell_amount_ttc !== null && l.sell_amount_ttc !== undefined) ? Number(l.sell_amount_ttc) : 0;
      return sell - cost;
    }

    function renderLines() {
      const tbody = document.getElementById('sim-lines-body');
      const lines = state.currentLines || [];

      let tCost = 0, tRev = 0;
      let negativeFound = false;

      tbody.innerHTML = lines.map((l, i) => {
        const cost = (l.cost_amount_ttc !== null && l.cost_amount_ttc !== undefined) ? Number(l.cost_amount_ttc) : 0;
        const sell = (l.sell_amount_ttc !== null && l.sell_amount_ttc !== undefined) ? Number(l.sell_amount_ttc) : 0;

        const margin = sell - cost;
        const pct = sell > 0 ? (margin / sell) * 100 : 0;

        // totals (treat null as 0)
        tCost += cost;
        tRev += sell;

        const risk = isLineRisk(l);
        if (risk) negativeFound = true;

        let kpiClass = 'obs-fair';
        let kpiText = 'FAIR';
        if (sell <= 0 && cost > 0) { kpiClass = 'obs-poor'; kpiText = 'COST ONLY'; }
        else if (pct < 10) { kpiClass = 'obs-poor'; kpiText = 'POOR'; }
        else if (pct >= 35) { kpiClass = 'obs-excel'; kpiText = 'EXCEL'; }
        else if (pct >= 20) { kpiClass = 'obs-good'; kpiText = 'GOOD'; }

        const disabled = (state.isLocked || !AUTH.canEditDraft) ? 'disabled' : '';

        const type = (String(l.source_type || 'COSTING').toUpperCase() === 'EXTRA') ? 'EXTRA' : 'COST';
        const typeBadge = type === 'EXTRA'
          ? `<span class="badge bg-info text-dark">EXTRA</span>`
          : `<span class="badge bg-secondary">COST</span>`;

        // Description pick
        const desc = l.description_used || l.description_en || l.description_fr || l.charge_code || '—';
        const code = l.charge_code || '—';

        // delete allowed only for EXTRA (and only if editable)
        const canDelete = (type === 'EXTRA' && !disabled);
        const delBtn = canDelete
          ? `<button class="btn btn-sm btn-outline-danger" onclick="event.stopPropagation(); deleteExtraLine(${i});"><i class="fa-solid fa-trash"></i></button>`
          : `<span class="text-muted">—</span>`;

        return `
          <tr class="${risk ? 'table-warning' : ''}">
            <td class="text-muted small">${i+1}</td>
            <td class="small">
              <div class="fw-bold text-dark">${escapeHtml(desc)}</div>
              <div class="text-muted font-mono" style="font-size:0.7rem">${escapeHtml(code)}</div>
            </td>
            <td class="text-end font-mono text-muted small">${fmtMoney(cost)}</td>
            <td class="p-1">
              <input type="number" class="form-control form-control-sm input-sell" id="sell-${i}"
                value="${sell || ''}"
                placeholder="${cost > 0 ? 'Leave empty if not billed' : 'Enter TTC'}"
                oninput="updateLine(${i}, 'sell_amount_ttc', this.value)"
                ${disabled}>
            </td>
            <td class="text-end font-mono fw-bold small ${margin<0 ? 'text-danger' : 'text-success'}">${fmtMoney(margin)}</td>
            <td class="text-center">
              <span class="obs-pill ${kpiClass}">${kpiText}${sell>0 ? ` (${pct.toFixed(0)}%)` : ''}</span>
            </td>
            <td class="p-1">
              <input type="text" class="form-control form-control-sm smart-input" id="note-${i}"
                placeholder="Client-facing remarks..."
                value="${escapeAttr(l.remarks_client_facing || '')}"
                oninput="updateLine(${i}, 'remarks_client_facing', this.value)"
                ${disabled}>
            </td>
            <td class="text-center">${typeBadge}</td>
            <td class="text-center">${delBtn}</td>
          </tr>
        `;
      }).join('');

      // Risk UI
      const riskEl = document.getElementById('risk-warning');

    function updateSnapshot(cost, rev, riskFlag) {
      const margin = Number(rev || 0) - Number(cost || 0);
      const pct = rev > 0 ? (margin / rev * 100) : 0;

      const pctDisplay = document.getElementById('global-margin-pct');
      const badge = document.getElementById('global-margin-badge');

      pctDisplay.innerText = pct.toFixed(1) + "%";

      let bClass = 'bg-secondary', bText = 'WAITING';
      if (rev > 0) {
        if (pct < 10) { bClass = 'bg-danger'; bText = 'CRITICAL'; }
        else if (pct < 20) { bClass = 'bg-warning text-dark'; bText = 'LOW'; }
        else { bClass = 'bg-success'; bText = 'HEALTHY'; }
      }
      badge.innerHTML = `<span class="badge ${bClass} px-3 py-2">${bText}</span>`;

      const max = Math.max(cost, rev, Math.abs(margin)) || 100;
      const hCost = (cost / max) * 100;
      const hRev = (rev / max) * 100;
      const hProf = (Math.abs(margin) / max) * 100;

      document.getElementById('chart-cost-bar').style.height = `${hCost}%`;
      document.getElementById('chart-rev-bar').style.height = `${hRev}%`;
      document.getElementById('chart-prof-bar').style.height = `${hProf}%`;

      document.getElementById('chart-cost-val').innerText = (cost / 1000000).toFixed(1) + 'M';
      document.getElementById('chart-rev-val').innerText = (rev / 1000000).toFixed(1) + 'M';
      document.getElementById('chart-prof-val').innerText = (margin / 1000000).toFixed(1) + 'M';

      document.getElementById('footer-count').innerText = (state.currentLines || []).length;

      // risk chip toggle
      const chipRisk = document.getElementById('chip-risk');
      if (riskFlag) chipRisk.classList.remove('d-none'); else chipRisk.classList.add('d-none');
    }

    // ---- Utilities: Number to words (simple)
    function toWords(n) {
      const string = n.toString(),
            units = ['', 'one', 'two', 'three', 'four', 'five', 'six', 'seven', 'eight', 'nine', 'ten', 'eleven', 'twelve', 'thirteen', 'fourteen', 'fifteen', 'sixteen', 'seventeen', 'eighteen', 'nineteen'];
      const tens = ['', '', 'twenty', 'thirty', 'forty', 'fifty', 'sixty', 'seventy', 'eighty', 'ninety'];
      const scales = ['', 'thousand', 'million', 'billion', 'trillion', 'quadrillion', 'quintillion'];

      if (n === 0) return 'zero';

      let start = string.length;
      let chunks = [];
      while (start > 0) {
        const end = start;
        chunks.push(string.slice((start = Math.max(0, start - 3)), end));
      }

      let chunksLen = chunks.length;
      if (chunksLen > scales.length) return '';

      let words = [], word;

      for (let i = 0; i < chunksLen; i++) {
        let chunk = parseInt(chunks[i], 10);
        if (chunk) {
          let ints = chunks[i].split('').reverse().map(x => parseInt(x, 10));
          if (ints[1] === 1) ints[0] += 10;
          if ((word = scales[i])) words.push(word);
          if ((word = units[ints[0]])) words.push(word);
          if ((word = tens[ints[1]])) words.push(word);
          if ((word = units[ints[2]])) words.push(word + ' hundred');
        }
      }
      return words.reverse().join(' ').toUpperCase();
    }

    function escapeHtml(s) {
      return String(s ?? '').replace(/[&<>"']/g, (ch) => ({
        '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#039;'
      }[ch]));
    }
    function escapeAttr(s) {
      return escapeHtml(s).replace(/"/g, '&quot;');
    }
